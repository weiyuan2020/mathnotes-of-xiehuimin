	\chapter{归纳方法}
	一个猜想性的一般命题若能在新特例中得到证实，则变得更加可信。\\
	\begin{enumerate}
		\item 随时准备修正我们的信念
		\item 有充分理由使我们改变信念，则应改变
		\item 无充分理由，不应轻易改变信念
	\end{enumerate}
	
	例题：
	
	1. 找规律
	\begin{gather*}
		11, 31, 41, 61, 71, 101, 131, \dots
	\end{gather*}
	以上数字都是素数
	\begin{gather*}
		21 = 3 \times 7\\
		51 = 3 \times 17\\
		81 = 3^4\\
		91 = 7 \times 13\\
		111 = 3 \times 37 \\
		121 = 11^2 \\
		141 = 3 \times 47\\
	\end{gather*}

	2. 
	\begin{gather*}
		\begin{aligned}
		&1			&=&0+1	\\
		&2+3+4		&=&1+8	\\
		&5+6+7+8+9	&=&8+27	\\
		&10+11+\dots+16	&=&27+64 \\
		\dots \\
		&n^2+1 + \dots + (n+1)^2 &=& n^3+(n+1)^3\\
		\end{aligned}
	\end{gather*}
	
	3.
	\begin{gather*}
		1,1+3,1+3+5,1+3+5+7,\dots
	\end{gather*}
	规律\\
	\begin{gather*}
		1+3+\dots +(2n-1) = \frac{(1+2n-1)\times n}{2} = n^2
	\end{gather*}
	
	4.
	\begin{gather*}
		1,1+8,1+8+27,1+8+27+64, \dots
	\end{gather*}
	\begin{gather*}
		1^3+2^3+\dots + n^3 = (1+2+\dots+n)^2 = \frac{n^2(n+1)^2}{4}
	\end{gather*}
