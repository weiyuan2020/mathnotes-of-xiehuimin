
\section{The extended real number system}
\mybox{使用正负无穷大扩充实数系}
\begin{myDef}
    \label{thm:1.23}
    The extended real number system consists of the real field $\R$
    and two symbols, $+\infty$ and $-\infty$. 
    We preserve the original order in $\R$, 
    and define
    \begin{equation*}
        -\infty <x<+\infty
    \end{equation*}
    for every $x\in \R$
\end{myDef}

\begin{asparaenum}[(a)]
\item If $x$ is real then
\begin{equation*}
    x+\infty = +\infty,\quad
    x-\infty = -\infty,\quad
    \frac{x}{+\infty}=\frac{x}{-\infty}=0.
\end{equation*}

\item If $x>0$ 
then $x\cdot (+\infty) = +\infty$, 
$x\cdot(-\infty)=-\infty$.

\item If $x<\infty$ 
then $x\cdot(+\infty)=-\infty$, 
$x\cdot(-\infty)= +\infty$.
\end{asparaenum}
