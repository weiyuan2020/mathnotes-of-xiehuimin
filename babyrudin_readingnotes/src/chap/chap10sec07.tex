\section{Stokes' theorem}

\begin{thm}
    \label{thm:10.33}
    If $\Psi$ is a $k$-chain of class $\mathscr{C}''$ in an open set $V \subset \R^m$ and if $\omega$ is a $(k-1)$-form of class $\mathscr{C}'$ in $V$, then 
    \begin{equation}
        \label{eq:10.91}
        \int_{\Psi} \d \omega = 
        \int_{\partial \Psi} \omega .
    \end{equation}
\end{thm}
